Ⅰ 704 二分查找

note

1. 二分法的两种区间划分

每次写二分最容易纠结的part就是下面C++代码里面的A,B,C,D行:

  • A 处是用nums.size()-1还是nums.size()?
  • B 处是 <= 还是 < ?
  • C 处是 mid 还是 mid - 1 ?
  • D 处是 mid 还是 mid + 1 ?

实际上这些不同的本质实际上是对于数组区间的划分法不同：

左闭右闭区间 左闭右开区间
A high=nums.size()-1 high=nums.size()
B while(low<=high) while(low<high)
C high=mid-1; high=mid;
D low=mid+1; low=mid+1;

2. 二分法取中间值防溢出

 mid = (low+high) / 2

E 处的写法在两大数相加的时候容易发生溢出，可以改成下面的写法：

 mid = low + (( high - low ) / 2)

C++ solution

class Solution {
public 
    int search(vector<int>& nums, int target) {

/*A*/   int low=0,high=nums.size()-1;

/*B*/   while(low<=high){

/*E*/        int mid=(low+high)/2;

            if(target==nums[mid]){
                return mid;
            }else if(target<nums[mid]){
/*C*/           high=mid-1;
            }else{
/*D*/           low=mid+1;
            }

        }

        return -1;
    }
};

Java solution

class Solution {
    public int search(int[] nums, int target) {

        int low=0,high=nums.length-1;

        while(low<=high){

            int mid=(low+high)/2;

            if(nums[mid]==target){
                return mid;
            }else if(nums[mid]>=target){
                high=mid-1;
            }else{
                low=mid+1;
            }

        }

        return -1;

    }
}

Ⅱ 35. 搜索插入位置

C++ solution

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int low =0,high=nums.size()-1,mid,ans=nums.size();

        while(low<=high){

            mid =(low+high)>>1;
            
            if(target<=nums[mid]){//在左边 
                ans=mid;
                high=mid-1;
            }else{
                low=mid+1;
            }
        }
        return ans;

    }
};

Java solution


class Solution {
    public int searchInsert(int[] nums, int target) {

        int low = 0, high = nums.length-1, ans = nums.length;

        while(low<=high){

            int mid = ( low + high ) >> 1;

            if(target <= nums[mid]){// 在左边

                ans = mid;

                high = mid-1;

            } else{

                low = mid +1;

            }
        }
         return ans;   
    }
}

Ⅲ 34. 在排序数组中查找元素的第一个和最后一个位置

C++ solution


class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        
        int left = binarySearch(nums,target,true);
        int right =  binarySearch(nums,target,false)-1;
        
        if(left <= right && right < nums.size() && nums[left] == target && nums[right] == target){
            
            return vector<int>{left,right};
        }else{
            return vector<int>{-1,-1};
        }
        
    }
    
    int binarySearch(vector<int>nums,int target,bool isLeft){
        int low =0,high=nums.size()-1,mid,ans=nums.size();
        
        while(low<=high){
            
            mid=(low+high)>>1;
            
            if(target<nums[mid]||isLeft&&target<=nums[mid]){
                ans = mid;
                high = mid - 1;
                }else{				
                low = mid + 1;
            }
        }
        return ans;
    }
    
};
