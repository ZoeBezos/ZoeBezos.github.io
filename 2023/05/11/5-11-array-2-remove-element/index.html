class Solution{​ public:    vector sortedSquares(vector& nums){        int len = nums.size();        vectorans(len);​        int l=0;        int r=len-1;        int idx=len-1;​        while(idx>=0){            if(nums[l]*nums[l]>nums[r]*nums[r]){                ans[idx]=nums[l]*nums[l];                l++;           }else{                ans[idx]=nums[r]*nums[r];                r–;           }            idx–;       }​        return ans;   }​};​c++
